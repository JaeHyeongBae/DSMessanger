442  //update chat log textbox with latest friend selection & messages
443 private: void updateChatLog()
444 {
445  if (friend_listbox->SelectedIndex >= 0) {
446  MemberType* mem = app->memberList.RetrieveByName(msclr::interop::marshal_as<std::string>(friend_listbox->SelectedItem->ToString()));
447  if (mem == nullptr)
448  {
449  refreshFriendList();
450  chatlog_richtextbox->Text = "";
451  return;
452  }
453  String^ log = gcnew String("");
454  my_mem->chatList.ResetList();
455  for (int i = 0; i < my_mem->chatList.GetLength(); i++)
456  {
457  MessageType msg;
458  my_mem->chatList.GetNextItem(msg);
459  if (msg.reciever == mem->ID || msg.sender == mem->ID)
460  log += gcnew String((msg.sender + ">> " + msg.message + "\n").c_str());
461  }
462  chatlog_richtextbox->Text = log;
463  }
464  else
465  chatlog_richtextbox->Text = "";
466 }
